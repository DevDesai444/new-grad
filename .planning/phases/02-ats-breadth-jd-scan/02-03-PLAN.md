Function signature (LOCKED):
    def extract_experience_range(description: str) -> tuple[int | None, int | None]

Returns: (experience_min, experience_max) — either or both may be None.
  - Exact range:        "0-3 years"     -> (0, 3)
                        "3 to 5 years"  -> (3, 5)
                        "5–7 years"     -> (5, 7)        (en-dash variant)
                        "5-7 years"     -> (5, 7)
  - Open-ended minimum: "5+ years"      -> (5, None)
                        "5+ yrs"        -> (5, None)
  - Entry signals:      "entry-level"          -> (0, None)
                        "entry level"          -> (0, None)
                        "recent graduate"      -> (0, None)
                        "no experience required" -> (0, None)
                        "0-N years" (any N)    -> (0, N)
  - No match:           anything else -> (None, None)

Algorithm (LOCKED per <specifics>):
    1. If description is None or empty → return (None, None).
    2. Normalize: description = description[:5000]   (cap to avoid pathological regex on 100KB HTML)
                  Lowercase for matching only.
    3. Run the RANGE_PATTERN first (numeric ranges take precedence — they're more specific):
         RANGE_PATTERN = re.compile(
             r"\b(\d+)\+?\s*(?:to|-|–|—)\s*(\d+)\s*\+?\s*years?\b",
             re.IGNORECASE
         )
       First match wins; (int(group1), int(group2)) returned.
    4. If no range, try OPEN_MIN_PATTERN:
         OPEN_MIN_PATTERN = re.compile(
             r"\b(\d+)\+\s*years?\b",
             re.IGNORECASE
         )
       First match → (int(group1), None).
    5. If no open-min, try ENTRY_PATTERN (any of):
         ENTRY_PATTERN = re.compile(
             r"\b(?:entry[- ]?level|recent\s+graduate|no\s+experience\s+required|new\s+grad(?:uate)?)\b",
             re.IGNORECASE
         )
       First match → (0, None).
    6. No match → (None, None).

        ```python
        Endpoint: POST https://jobs.apple.com/api/role/search
        Auth: none (public).
        Dedup key: `apple:<positionId>` (NO per-company prefix per CONTEXT.md D-01a —

        cap of 25 pages, sort-monotonicity sanity fallback. Mirrors WorkdayAdapter from Plan 02-02.

        Per CONTEXT.md `<specifics>`: request body is the minimal form
            {"query": "", "locale": "en-us", "page": N, "pageSize": 20, "sort": {...}, "filters": {}}
        — may need live validation; lock here and document in adapter comments.
        """
        from __future__ import annotations

        import logging
        from typing import ClassVar
        from urllib.parse import urlparse

        import httpx

        from src.adapters.base import Adapter, SchemaDrift, SiteBlocked
        from src.models import CompanyConfig, RawPosting

        _HOST = "jobs.apple.com"
        _API_URL = "https://jobs.apple.com/api/role/search"
        _PAGE_SIZE = 20
        _COLD_START_CAP_PAGES = 25     # 25 pages × 20 postings = 500 first-run safety cap
        _TIMEOUT_S = 30.0
        _UA = "Mozilla/5.0 (compatible; new-grad-tracker; +https://github.com/DevDesai444/new-grad)"

        logger = logging.getLogger(__name__)


        class AppleAdapter(Adapter):
            name: ClassVar[str] = "apple"

            @classmethod
            def matches(cls, url: str) -> bool:
                try:
                    host = urlparse(url).hostname or ""
                except ValueError:
                    return False
                return host.lower() == _HOST

            def _build_body(self, page: int) -> dict:
                return {
                    "query": "",
                    "locale": "en-us",
                    "page": page,
                    "pageSize": _PAGE_SIZE,
                    "sort": {"field": "postingDate", "order": "desc"},
                    "filters": {},
                }

            def _extract_postings_array(self, payload, company: CompanyConfig) -> list[dict]:
                if not isinstance(payload, dict):
                    raise SchemaDrift(
                        f"Apple {company.name}: response is not a JSON object "
                        f"(got {type(payload).__name__})"
                    )
                # Newer shape: `results`. Older shape: `searchResults`. Coalesce.
                array = payload.get("results")
                if array is None:
                    array = payload.get("searchResults")
                if array is None:
                    raise SchemaDrift(
                        f"Apple {company.name}: response missing both 'results' and "
                        f"'searchResults' keys (got: {list(payload.keys())})"
                    )
                if not isinstance(array, list):
                    raise SchemaDrift(
                        f"Apple {company.name}: postings array is not a list "
                        f"(got {type(array).__name__})"
                    )
                return array

            def _extract_position_id(self, posting: dict) -> str | None:
                # Coalesce — Apple uses both `positionId` and `id` in different response shapes.
                pid = posting.get("positionId") or posting.get("id")
                if pid is None:
                    return None
                pid_str = str(pid).strip()
                return pid_str or None

            def _parse_posted_date_for_sanity(self, posting: dict) -> str | None:
                # Used ONLY by D-04 sort-monotonicity check; returns the raw string.
                return posting.get("postingDate") or posting.get("postDateInGMT")

            def fetch(
                self,
                company: CompanyConfig,
                seen_keys: set[str] | None = None,
            ) -> list[RawPosting]:
                """D-04 paginated fetch.

                Pagination semantics (sort=desc by postingDate, newest first):
                  - Fetch page 0; check sort-monotonicity vs prior page's last postingDate
                    (no prior page on page 0 — track tail).
                  - If seen_keys is provided and the LAST posting on this page is in seen_keys,
                    stop (early-termination — older entries are known).
                  - If total <= (page+1)*pageSize OR returned page is short, stop (no more results).
                  - If page_count >= _COLD_START_CAP_PAGES, stop (cold-start safety).
                  - If sort-monotonicity violated (page N+1's first newer than page N's last),
                    log warning and stop paginating beyond the cap.
                """
                seen_keys = seen_keys or set()
                result: list[RawPosting] = []
                prev_page_tail_date: str | None = None
                sort_violation_logged = False

                for page in range(_COLD_START_CAP_PAGES):
                    body = self._build_body(page)
                    try:
                        response = httpx.post(
                            _API_URL,
                            json=body,
                            headers={"User-Agent": _UA, "Accept": "application/json"},
                            timeout=_TIMEOUT_S,
                        )
                    except httpx.HTTPError:
                        raise

                    if response.status_code in (403, 429):
                        raise SiteBlocked(
                            f"Apple {company.name}: HTTP {response.status_code} from {_API_URL}"
                        )
                    if response.status_code >= 500:
                        raise SiteBlocked(
                            f"Apple {company.name}: HTTP {response.status_code} (server error) "
                            f"from {_API_URL}"
                        )
                    response.raise_for_status()

                    try:
                        payload = response.json()
                    except ValueError as e:
                        raise SchemaDrift(
                            f"Apple {company.name}: response body is not JSON"
                        ) from e

                    postings_array = self._extract_postings_array(payload, company)

                    if not postings_array:
                        # Empty page — no more results.
                        break

                    # Sort-monotonicity check (D-04): if this page's FIRST posting is NEWER
                    # than the prev page's LAST posting, sort was ignored.
                    if prev_page_tail_date is not None:
                        first_date = self._parse_posted_date_for_sanity(postings_array[0])
                        if (
                            first_date is not None
                            and prev_page_tail_date is not None
                            and first_date > prev_page_tail_date  # ISO-8601 string lexical compare works for UTC
                        ):
                            if not sort_violation_logged:
                                logger.warning(
                                    "Apple %s: sort order monotonicity violated on page %d "
                                    "(sort param may be ignored by API). Falling back to 25-page cap.",
                                    company.name, page,
                                )
                                sort_violation_logged = True
                            # Don't stop immediately — fall back to the cold-start cap.

                    page_keys_appended: list[str] = []
                    for posting in postings_array:
                        if not isinstance(posting, dict):
                            continue
                        position_id = self._extract_position_id(posting)
                        if position_id is None:
                            continue
                        dedup_key = f"apple:{position_id}"
                        enriched = dict(posting)
                        enriched["__dedup_key"] = dedup_key
                        enriched["__position_id"] = position_id
                        result.append(
                            RawPosting(
                                source_company=company.name,
                                source_adapter=self.name,
                                raw=enriched,
                            )
                        )
                        page_keys_appended.append(dedup_key)

                    # Update sort-monotonicity tracker BEFORE early-termination check.
                    prev_page_tail_date = self._parse_posted_date_for_sanity(postings_array[-1])

                    # D-04 early-termination: last key on this page already in seen.json.
                    if (
                        not sort_violation_logged
                        and page_keys_appended
                        and page_keys_appended[-1] in seen_keys
                    ):
                        break

                    # Short page → no more results.
                    if len(postings_array) < _PAGE_SIZE:
                        break

                return result
        ```

          ```python
          _RANGE_PATTERN = re.compile(

              return (None, None)
          ```
