Endpoint: <API URL pattern>
Auth: none (public ATS).
Dedup key: `<prefix>:<identifier>:<posting-id>` — extracted from the API response per
PITFALLS.md Pitfall 5 (stable IDs; never URL-based).

Phase 2 D-03 — happy path + 4 error-path tests + generic-propagation test ship with this module.

from __future__ import annotations

from typing import ClassVar
from urllib.parse import urlparse

import httpx

from src.adapters.base import Adapter, SchemaDrift, SiteBlocked
from src.models import CompanyConfig, RawPosting

_HOST = "<hostname>"
_TIMEOUT_S = 20.0


class <Name>Adapter(Adapter):
    name: ClassVar[str] = "<name>"

    @classmethod
    def matches(cls, url: str) -> bool:
        try:
            host = urlparse(url).hostname or ""
        except ValueError:
            return False
        return host.lower() == _HOST

    @staticmethod
    def _extract_identifier(url: str) -> str:
        parsed = urlparse(url)
        segments = [s for s in parsed.path.split("/") if s]
        if not segments:
            raise ValueError(f"Cannot extract <Name> identifier from URL: {url}")
        return segments[0]

    def fetch(self, company: CompanyConfig) -> list[RawPosting]:
        identifier = self._extract_identifier(company.url)
        api_url = f"<API URL with {identifier}>"

        try:
            response = httpx.get(api_url, timeout=_TIMEOUT_S)
        except httpx.HTTPError:
            raise

        if response.status_code in (403, 429):
            raise SiteBlocked(f"<Name> {company.name}: HTTP {response.status_code} from {api_url}")
        if response.status_code >= 500:
            raise SiteBlocked(f"<Name> {company.name}: HTTP {response.status_code} (server error) from {api_url}")
        response.raise_for_status()

        try:
            payload = response.json()
        except ValueError as e:
            raise SchemaDrift(f"<Name> {company.name}: response body is not JSON") from e

            enriched["__identifier"] = identifier
            result.append(
                RawPosting(
