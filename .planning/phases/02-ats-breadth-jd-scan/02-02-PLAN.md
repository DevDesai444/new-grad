```python
import re
from typing import NamedTuple

# D-01 (CONTEXT.md <specifics>): tenant + wd-number + optional locale + site.
_WORKDAY_URL_RE = re.compile(
    r"^https?://"
    r"(?P<tenant>[a-z0-9-]+)\.wd(?P<wd_num>\d+)\.myworkdayjobs\.com/"
    r"(?:[a-z]{2}-[A-Z]{2}/)?"
    r"(?P<site>[A-Za-z0-9_-]+)/?$",
)


class WorkdayURLParts(NamedTuple):
    tenant: str
    wd_num: str
    site: str


def _parse_workday_url(url: str) -> WorkdayURLParts:
    """Parse a Workday careers URL into (tenant, wd_num, site).

    Raises SchemaDrift naming WHICH piece is missing if regex fails (defensive: a partial-match
    diagnostic helps the user fix the URL without log-diving).
    """
    m = _WORKDAY_URL_RE.match(url.strip())
    if m is None:
        # Try to produce a more-specific diagnostic by partial probing.
        bare_re = re.compile(r"^https?://([a-z0-9-]+)\.wd(\d+)\.myworkdayjobs\.com/")
        bm = bare_re.match(url.strip())
        if bm is None:
            raise SchemaDrift(
                f"Workday URL did not match expected pattern (missing tenant.wdN.myworkdayjobs.com host): {url}"
            )
        # Host matched but path/site didn't — tell the user which piece is missing.
        raise SchemaDrift(
            f"Workday URL did not match expected pattern (tenant + wd-number parsed but site segment is missing or malformed): {url}"
        )
    return WorkdayURLParts(
        tenant=m.group("tenant"),
        wd_num=m.group("wd_num"),
        site=m.group("site"),
    )
```

<reference_pattern_for_postedon_resolver>
<!-- Multi-form postedOn parsing — runs inside the adapter, stashes UTC datetime in __posted_date_utc -->

```python
import re
from datetime import datetime, timedelta, UTC

_RELATIVE_PATTERNS = (
    (re.compile(r"^posted\s+today\s*$", re.IGNORECASE),                lambda m, now: now),
    (re.compile(r"^posted\s+yesterday\s*$", re.IGNORECASE),            lambda m, now: now - timedelta(days=1)),
    (re.compile(r"^posted\s+(\d+)\+?\s*days?\s+ago\s*$", re.IGNORECASE), lambda m, now: now - timedelta(days=int(m.group(1)))),
)


def _parse_workday_posted(value, run_started_at: datetime):
    """Parse Workday `postedOn` (epoch ms int | ISO string | relative string) → UTC datetime | None."""
    if value is None or value == "":
        return None
    # Form 1: epoch milliseconds (int or float).
    if isinstance(value, (int, float)):
        try:
            return datetime.fromtimestamp(float(value) / 1000.0, tz=UTC)
        except (ValueError, OSError, OverflowError):
            return None
    if not isinstance(value, str):
        return None
    s = value.strip()
    # Form 2: ISO-8601 string.
    iso = _parse_iso_to_utc(s)    # import from normalizer
    if iso is not None:
        return iso
    # Form 3-6: relative string.
    for pattern, fn in _RELATIVE_PATTERNS:
        m = pattern.match(s)
        if m is not None:
            return fn(m, run_started_at)
    # Form 7: unknown — leave None.
    return None
```

          ```python
                  return self._fetch_page_and_emit(api_url, parts, company, offset=0, run_started_at=datetime.now(UTC))

              def _fetch_page_and_emit(
                  self,
                  api_url: str,
                  parts: WorkdayURLParts,
                  company: CompanyConfig,
                  offset: int,
                  run_started_at: datetime,
              ) -> list[RawPosting]:
                  body = {"limit": _PAGE_SIZE, "offset": offset, "searchText": "", "appliedFacets": {}}
                  headers = {
                      "Content-Type": "application/json",
                      "Accept": "application/json",
                      "User-Agent": _USER_AGENT,
                  }
                  try:
                      response = httpx.post(api_url, json=body, headers=headers, timeout=_TIMEOUT_S)
                  except httpx.HTTPError:
                      raise   # ADP-12: orchestrator's per-company catch logs type+str

                  if response.status_code in (403, 429):
                      raise SiteBlocked(f"Workday {company.name}: HTTP {response.status_code} from {api_url}")
                  if response.status_code >= 500:
                      raise SiteBlocked(f"Workday {company.name}: HTTP {response.status_code} (server error) from {api_url}")
                  response.raise_for_status()

                  try:
                      payload = response.json()
                  except ValueError as e:
                      raise SchemaDrift(f"Workday {company.name}: response body is not JSON") from e

                  if not isinstance(payload, dict) or "jobPostings" not in payload:
                      got = list(payload.keys()) if isinstance(payload, dict) else type(payload).__name__
                      raise SchemaDrift(f"Workday {company.name}: missing 'jobPostings' key (got: {got})")
                  if not isinstance(payload["jobPostings"], list):
                      raise SchemaDrift(
                          f"Workday {company.name}: 'jobPostings' is not a list "
                          f"(got {type(payload['jobPostings']).__name__})"
                      )

                  result: list[RawPosting] = []
                  for job in payload["jobPostings"]:
                      if not isinstance(job, dict):
                          continue
                      # Dedup id: prefer bulletFields[0] (JOB_REQ_ID); fall back to externalPath suffix.
                      job_id = None
                      bullets = job.get("bulletFields")
                      if isinstance(bullets, list) and bullets:
                          job_id = str(bullets[0]).strip() or None
                      if not job_id:
                          ext = job.get("externalPath") or ""
                          if "/" in ext:
                              job_id = ext.rstrip("/").split("/")[-1] or None
                      if not job_id:
                          continue   # skip malformed entry

                      ext_path = job.get("externalPath") or ""
                      full_url = f"https://{parts.tenant}.wd{parts.wd_num}.myworkdayjobs.com{ext_path}"
                      dedup_key = f"wd:{parts.tenant}:{job_id}"
                      posted_dt = _parse_workday_posted(job.get("postedOn"), run_started_at)

                      enriched = dict(job)
                      enriched["__dedup_key"] = dedup_key
                      enriched["__tenant"] = parts.tenant
                      enriched["__posting_url"] = full_url
                      enriched["__posted_date_utc"] = posted_dt   # consumed by normalizer (Task 3 of this plan adds the dispatch entry)
                      result.append(
                          RawPosting(
                              source_company=company.name,
                              source_adapter=self.name,
                              raw=enriched,
                          )
                      )
                  return result
          ```
